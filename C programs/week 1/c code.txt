#include<stdio.h>
int main(){
    int day;
    for(int i=1;i>=1;i++){
        scanf("%d/n",&day);
        if(day>=1 && day<=7){
            break;}
            
        else{
            printf("Invalid input");
        }
        }
        switch(day) {
        case 1: printf("Monday\n"); 
        break;
        case 2: printf("Tuesday\n");
        break;
        case 3: printf("Wednesday\n");
        break;
        case 4: printf("Thursday\n");
        break;
        case 5: printf("Friday\n");
        break;
        case 6: printf("Saturday\n");
        break;
        case 7: printf("Sunday\n");
        break;
    }
    
    return 0;
}
