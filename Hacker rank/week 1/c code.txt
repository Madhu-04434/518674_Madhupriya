1]  #include <stdio.h>
void plusMinus(int arr[], int n) {
    int pos = 0, neg = 0, zero = 0;
    for (int i = 0; i < n; i++) {
        if (arr[i] > 0) pos++;
        else if (arr[i] < 0) neg++;
        else zero++;
    }
    printf("%.6f\n%.6f\n%.6f\n", (float)pos/n, (float)neg/n, (float)zero/n);
}
int main() {
    int n;
    scanf("%d", &n);
    int arr[n];
    for (int i = 0; i < n; i++) scanf("%d", &arr[i]);
    plusMinus(arr, n);
    return 0;
}

void miniMaxSum(int arr[5]) {
    long total = 0;
    int min = arr[0], max = arr[0];
    for (int i = 0; i < 5; i++) {
        total += arr[i];
        if (arr[i] < min) min = arr[i];
        if (arr[i] > max) max = arr[i];
    }
    printf("%ld %ld", total - max, total - min);
}
int main() {
    int arr[5];
    for (int i = 0; i < 5; i++) {
        scanf("%d", &arr[i]);
    }
    miniMaxSum(arr);
    return 0;
}

#include <string.h>
int main() {
    char time[10];
    scanf("%s", time);
    int hour = (time[0] - '0') * 10 + (time[1] - '0');
    char period = time[8];
    if (period == 'P' && hour != 12) {
        hour += 12;
    } else if (period == 'A' && hour == 12) {
        hour = 0;
    }
    printf("%02d:%c%c:%c%c", hour, time[3], time[4], time[6], time[7]);
    return 0;
}

    char queries[q][100];
    for(int i=0; i<q; i++) {
        scanf("%s", queries[i]);
    }
    for(int i=0; i<q; i++) {
        int count = 0;
        for(int j=0; j<n; j++) {
            if(strcmp(queries[i], strings[j]) == 0) {
                count++;
            }
        }
        printf("%d\n", count);
    }

    return 0;
